on_death = {
	on_actions = {
		give_unpressed_admin_title_claim
	}
}

give_unpressed_admin_title_claim = {
	effect = {
		if = {
			limit = {
				is_independent_ruler = yes
				government_has_flag = government_is_administrative
				primary_title.tier >= tier_empire
			}
			primary_title = {
				save_temporary_scope_as = admin_title
			}
			scope:admin_title = {
				ordered_title_heir = {
					limit = {
						NOT = { has_claim_on = scope:admin_title }
					}
					order_by = "appointment_candidate_score(scope:admin_title)"
					check_range_bounds = no
					max = additional_succession_claims_total
					add_unpressed_claim = scope:admin_title
				}
			}
		}
	}
}
