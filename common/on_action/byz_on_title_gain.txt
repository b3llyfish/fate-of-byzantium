on_title_gain = {
	on_actions = {
		sultan_of_rum_trait
		bulgarian_resistance_trait
	}
}

sultan_of_rum_trait = {
	effect = {
		if = {
			limit = {
				exists = scope:previous_holder
				scope:previous_holder = {
					OR = {
						has_trait = sultan_rum_minor
						has_trait = sultan_rum
						has_trait = sultan_rum_major
					}
				}
				scope:title = title:k_rum
				OR = {
					religion = religion:islam_religion
					culture = { has_cultural_pillar = heritage_turkic }
				}
			}
			add_sultan_rum_trait_effect = yes
		}
	}
}

bulgarian_resistance_trait = {
	effect = {
		if = {
			limit = {
				exists = scope:previous_holder
				scope:previous_holder = {
					OR = {
						has_trait = bulgarian_resistance_minor
						has_trait = bulgarian_resistance
						has_trait = bulgarian_resistance_major
					}
				}
				scope:title = title:k_bulgaria
				culture = { has_cultural_pillar = heritage_south_slavic }
			}
			add_bulgarian_resistance_trait_effect = yes
		}
	}
}
