on_rank_up = {
	on_actions = {
		remove_sultan_rum_rank_up
	}
}

remove_sultan_rum_rank_up = {
	effect = {
		if = {
			limit = {
				scope:title = title:k_rum
				highest_held_title_tier >= tier_empire
				OR = {
					has_trait = sultan_rum_minor
					has_trait = sultan_rum
					has_trait = sultan_rum_major
				}
			}
			remove_sultan_rum_trait_effect = yes
		}
	}
}
