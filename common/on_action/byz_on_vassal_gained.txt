on_vassal_gained = {
	on_actions = {
		remove_sultan_of_rum_trait_vassal
	}
}

remove_sultan_of_rum_trait_vassal = {
	effect = {
		if = {
			limit = {
				scope:vassal = {
					has_title = title:k_rum
					OR = {
						has_trait = sultan_rum_minor
						has_trait = sultan_rum
						has_trait = sultan_rum_major
					}
				}
			}
			scope:vassal = {
				remove_sultan_rum_trait_effect = yes
			}
		}
	}
}
