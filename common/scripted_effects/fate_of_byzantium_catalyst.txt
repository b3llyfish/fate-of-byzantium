# scope: character owning the scheme
fate_of_byzantium_catalyst_challenge_status_scheme_effect = {
	if = {
		limit = {
			struggle:fate_of_byzantium ?= {
				phase_has_catalyst = catalyst_challenge_status_scheme
			}
			involved_in_byz_struggle = yes
		}
		save_scope_as = trigger_char
		struggle:fate_of_byzantium ?= {
			activate_struggle_catalyst = {
				catalyst = catalyst_challenge_status_scheme
				character = scope:trigger_char
			}
		}
	}
}
