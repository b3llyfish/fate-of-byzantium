admin_china_effect = {
	if = {
		limit = {
			has_game_rule = admin_china_on
			exists = title:h_china.holder
		}
		title:h_china.holder = {
			asia_admin_title_effect = yes
		}
		culture:han = {
			add_culture_tradition = tradition_ep3_cultivated_sophistication
		}
	}
}

asia_admin_title_effect = {
	if = {
		limit = {
			is_independent_ruler = yes
		}
		change_to_administrative_effect = yes
		every_vassal_or_below = {
			if = {
				limit = {
					government_has_flag = government_is_feudal
				}
				change_to_administrative_effect = yes
			}
		}
	}
}
