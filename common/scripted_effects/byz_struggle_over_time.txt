byz_struggle_yearly_effect = {
	# optimism
	if = {
		limit = {
			phase_has_catalyst = catalyst_byz_control_struggle_region_over_time
			phase_has_catalyst = catalyst_byz_major_control_struggle_region_over_time
		}
		if = {
			limit = {
				struggle_byzantium_emperor_struggle_region_county_count > struggle_byzantium_region_count_high
			}
			activate_struggle_catalyst = catalyst_byz_major_control_struggle_region_over_time
		}
		else_if = {
			limit = {
				struggle_byzantium_emperor_struggle_region_county_count > struggle_byzantium_region_count_low
			}
			activate_struggle_catalyst = catalyst_byz_control_struggle_region_over_time
		}
	}
	# stagnation
	# decline
	if = {
		limit = {
			OR = {
				phase_has_catalyst = catalyst_byz_loss_control_greek_region_over_time
				phase_has_catalyst = catalyst_byz_major_loss_control_greek_region_over_time
			}
		}
		if = {
			limit = {
				struggle_byzantium_emperor_greek_county_count < struggle_byzantium_region_greek_county_count_low
			}
			activate_struggle_catalyst = catalyst_byz_major_loss_control_greek_region_over_time
		}
		else_if = {
			limit = {
				struggle_byzantium_emperor_greek_county_count < struggle_byzantium_region_greek_county_count_high
			}
			activate_struggle_catalyst = catalyst_byz_loss_control_greek_region_over_time
		}
	}
}
